      {
          "name": "Launch",
          "type": "go",
          "request": "launch",
          "mode": "auto",
          "program": "${fileDirname}",
          "env": {},
          "args": []
      }
  ]
}

  • fork/exec d:\cache\go-build618440214\b001\exe\sitemap.exe: Access is denied.
